function hyStrNum(str)
        if str==nil then return 40 end
        local strn = nil
        for i=1,string.len(str) do
                local num=string.sub(str,i,i)
                if tonumber(num) then
                        if strn==nil then
                                strn=num
                        else
                                strn=strn..num
                        end
                end
        end
        return tonumber(strn)
end

                        pressHomeKey(1)
                        mSleep(500)
                else
                        unlockDevice() ---不是IOS10，那么直接解锁屏幕
                end
        end
end
